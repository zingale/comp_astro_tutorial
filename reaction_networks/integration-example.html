Integration Example¶

We can use pynucastro to generate the righthand side function for an astrophysical reaction network.

We’ll create a CNO network that has the rates for all 4 CNO cycles + hot-CNO, as listed here: https://reaclib.jinaweb.org/popularRates.php

files = ["c12-pg-n13-ls09",
         "c13-pg-n14-nacr",
         "n13--c13-wc12",
         "n13-pg-o14-lg06",
         "n14-pg-o15-im05",
         "n15-pa-c12-nacr",
         "o14--n14-wc12",
         "o15--n15-wc12",
         "n15-pg-o16-li10",
         "o16-pg-f17-ia08",
         "f17--o17-wc12",
         "o17-pa-n14-il10",
         "o17-pg-f18-il10",
         "f18--o18-wc12",
         "o18-pa-n15-il10",
         "o18-pg-f19-il10",
         "f19-pa-o16-nacr",
         "o14-ap-f17-Ha96c",
         "f17-pg-ne18-cb09",
         "ne18--f18-wc12",
         "f18-pa-o15-il10"]
rc = pyrl.RateCollection(files)

We can visualize the network and rates linking the nuclei

rc.plot()

pynucastro can write out the python code needed to evaluate the reaction rates

pynet = pyrl.PythonNetwork(files)
pynet.write_network("cno_integration_example.py")

@numba.njit()
def rhs_eq(t, Y, rho, T):

    ip = 0
    ihe4 = 1
    ic12 = 2
    ic13 = 3
    in13 = 4
    in14 = 5
    in15 = 6
    io14 = 7
    io15 = 8
    io16 = 9
    io17 = 10
    io18 = 11
    if17 = 12
    if18 = 13
    if19 = 14
    ine18 = 15
    nnuc = 16

    tf = Tfactors(T)

    lambda_p_c12__n13 = p_c12__n13(tf)
    lambda_p_c13__n14 = p_c13__n14(tf)
    lambda_n13__c13__weak__wc12 = n13__c13__weak__wc12(tf)
    lambda_p_n13__o14 = p_n13__o14(tf)
    lambda_p_n14__o15 = p_n14__o15(tf)
    lambda_p_n15__he4_c12 = p_n15__he4_c12(tf)
    lambda_o14__n14__weak__wc12 = o14__n14__weak__wc12(tf)
    lambda_o15__n15__weak__wc12 = o15__n15__weak__wc12(tf)
    lambda_p_n15__o16 = p_n15__o16(tf)
    lambda_p_o16__f17 = p_o16__f17(tf)
    lambda_f17__o17__weak__wc12 = f17__o17__weak__wc12(tf)
    lambda_p_o17__he4_n14 = p_o17__he4_n14(tf)
    lambda_p_o17__f18 = p_o17__f18(tf)
    lambda_f18__o18__weak__wc12 = f18__o18__weak__wc12(tf)
    lambda_p_o18__he4_n15 = p_o18__he4_n15(tf)
    lambda_p_o18__f19 = p_o18__f19(tf)
    lambda_p_f19__he4_o16 = p_f19__he4_o16(tf)
    lambda_he4_o14__p_f17 = he4_o14__p_f17(tf)
    lambda_p_f17__ne18 = p_f17__ne18(tf)
    lambda_ne18__f18__weak__wc12 = ne18__f18__weak__wc12(tf)
    lambda_p_f18__he4_o15 = p_f18__he4_o15(tf)

    dYdt = np.zeros((nnuc), dtype=np.float64)

    dYdt[ip] = (
       -rho*Y[ip]*Y[ic12]*lambda_p_c12__n13
       -rho*Y[ip]*Y[ic13]*lambda_p_c13__n14
       -rho*Y[ip]*Y[in13]*lambda_p_n13__o14
       -rho*Y[ip]*Y[in14]*lambda_p_n14__o15
       -rho*Y[ip]*Y[in15]*lambda_p_n15__he4_c12
       -rho*Y[ip]*Y[in15]*lambda_p_n15__o16
       -rho*Y[ip]*Y[io16]*lambda_p_o16__f17
       -rho*Y[ip]*Y[io17]*lambda_p_o17__he4_n14
       -rho*Y[ip]*Y[io17]*lambda_p_o17__f18
       -rho*Y[ip]*Y[io18]*lambda_p_o18__he4_n15
       -rho*Y[ip]*Y[io18]*lambda_p_o18__f19
       -rho*Y[ip]*Y[if19]*lambda_p_f19__he4_o16
       -rho*Y[ip]*Y[if17]*lambda_p_f17__ne18
       -rho*Y[ip]*Y[if18]*lambda_p_f18__he4_o15
       +rho*Y[ihe4]*Y[io14]*lambda_he4_o14__p_f17
       )

    dYdt[ihe4] = (
       -rho*Y[ihe4]*Y[io14]*lambda_he4_o14__p_f17
       +rho*Y[ip]*Y[in15]*lambda_p_n15__he4_c12
       +rho*Y[ip]*Y[io17]*lambda_p_o17__he4_n14
       +rho*Y[ip]*Y[io18]*lambda_p_o18__he4_n15
       +rho*Y[ip]*Y[if19]*lambda_p_f19__he4_o16
       +rho*Y[ip]*Y[if18]*lambda_p_f18__he4_o15
       )

    dYdt[ic12] = (
       -rho*Y[ip]*Y[ic12]*lambda_p_c12__n13
       +rho*Y[ip]*Y[in15]*lambda_p_n15__he4_c12
       )

    dYdt[ic13] = (
       -rho*Y[ip]*Y[ic13]*lambda_p_c13__n14
       +Y[in13]*lambda_n13__c13__weak__wc12
       )

    dYdt[in13] = (
       -Y[in13]*lambda_n13__c13__weak__wc12
       -rho*Y[ip]*Y[in13]*lambda_p_n13__o14
       +rho*Y[ip]*Y[ic12]*lambda_p_c12__n13
       )

    dYdt[in14] = (
       -rho*Y[ip]*Y[in14]*lambda_p_n14__o15
       +rho*Y[ip]*Y[ic13]*lambda_p_c13__n14
       +Y[io14]*lambda_o14__n14__weak__wc12
       +rho*Y[ip]*Y[io17]*lambda_p_o17__he4_n14
       )

    dYdt[in15] = (
       -rho*Y[ip]*Y[in15]*lambda_p_n15__he4_c12
       -rho*Y[ip]*Y[in15]*lambda_p_n15__o16
       +Y[io15]*lambda_o15__n15__weak__wc12
       +rho*Y[ip]*Y[io18]*lambda_p_o18__he4_n15
       )

    dYdt[io14] = (
       -Y[io14]*lambda_o14__n14__weak__wc12
       -rho*Y[ihe4]*Y[io14]*lambda_he4_o14__p_f17
       +rho*Y[ip]*Y[in13]*lambda_p_n13__o14
       )

    dYdt[io15] = (
       -Y[io15]*lambda_o15__n15__weak__wc12
       +rho*Y[ip]*Y[in14]*lambda_p_n14__o15
       +rho*Y[ip]*Y[if18]*lambda_p_f18__he4_o15
       )

    dYdt[io16] = (
       -rho*Y[ip]*Y[io16]*lambda_p_o16__f17
       +rho*Y[ip]*Y[in15]*lambda_p_n15__o16
       +rho*Y[ip]*Y[if19]*lambda_p_f19__he4_o16
       )

    dYdt[io17] = (
       -rho*Y[ip]*Y[io17]*lambda_p_o17__he4_n14
       -rho*Y[ip]*Y[io17]*lambda_p_o17__f18
       +Y[if17]*lambda_f17__o17__weak__wc12
       )

    dYdt[io18] = (
       -rho*Y[ip]*Y[io18]*lambda_p_o18__he4_n15
       -rho*Y[ip]*Y[io18]*lambda_p_o18__f19
       +Y[if18]*lambda_f18__o18__weak__wc12
       )

    dYdt[if17] = (
       -Y[if17]*lambda_f17__o17__weak__wc12
       -rho*Y[ip]*Y[if17]*lambda_p_f17__ne18
       +rho*Y[ip]*Y[io16]*lambda_p_o16__f17
       +rho*Y[ihe4]*Y[io14]*lambda_he4_o14__p_f17
       )

    dYdt[if18] = (
       -Y[if18]*lambda_f18__o18__weak__wc12
       -rho*Y[ip]*Y[if18]*lambda_p_f18__he4_o15
       +rho*Y[ip]*Y[io17]*lambda_p_o17__f18
       +Y[ine18]*lambda_ne18__f18__weak__wc12
       )

    dYdt[if19] = (
       -rho*Y[ip]*Y[if19]*lambda_p_f19__he4_o16
       +rho*Y[ip]*Y[io18]*lambda_p_o18__f19
       )

    dYdt[ine18] = (
       -Y[ine18]*lambda_ne18__f18__weak__wc12
       +rho*Y[ip]*Y[if17]*lambda_p_f17__ne18
       )

    return dYdt
